float  conversion (float lat , float lon ) 
{
	lat = lat * (180 / M_PI);
	lon  = lon * (180 / M_PI);
	float x = sin(M_PI/2-lat) * cos(lon);
  float y = sin( M_PI/2-lat) * sin(lon);
  float z = cos( M_PI/2-lat);
	float R=sqrt(x*x+y*y+z*z);
	return R ;
};
